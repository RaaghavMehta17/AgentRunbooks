1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.gateway.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "ops-agents.fullname" . }}-gateway)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.gateway.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "ops-agents.fullname" . }}-gateway'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "ops-agents.fullname" . }}-gateway --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.gateway.port }}
{{- else if contains "ClusterIP" .Values.service.gateway.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "ops-agents.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=gateway" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8000 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8000:$CONTAINER_PORT
{{- end }}

2. Check the health endpoint:
   kubectl exec -n {{ .Release.Namespace }} deploy/{{ include "ops-agents.fullname" . }}-gateway -- curl -s http://localhost:8000/healthz

3. Secrets Management:
   IMPORTANT: The secrets in this chart are empty by default. In production, use:
   - External Secrets Operator (https://external-secrets.io)
   - Sealed Secrets (https://github.com/bitnami-labs/sealed-secrets)
   - HashiCorp Vault
   
   To set secrets manually:
   kubectl create secret generic {{ include "ops-agents.fullname" . }}-secrets \
     --from-literal=GITHUB_TOKEN='your-token' \
     --from-literal=OPENAI_API_KEY='your-key' \
     --dry-run=client -o yaml | kubectl apply -f -

4. View logs:
   kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/component=gateway --tail=100
   kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/component=orchestrator --tail=100
